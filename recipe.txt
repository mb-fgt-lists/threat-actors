#####
# Workaround for Fortinet stupidities
#####



## BLOCK PARKLOGIC.COM FROM BEING RESOLVED
config system dns-database
    edit "parklogic.com"
        set domain "parklogic.com"
        set ttl 300
        config dns-entry
            edit 1
                set type NS
                set hostname "ns"
            next
            edit 2
                set hostname "ns"
                set ip 127.0.0.1
            next
        end
    next
end

## IF RESOLVED, NULL-ROUTE PARKLOGIC VIA ADDRESS
config firewall address
    edit "wildcard.parklogic.com"
        set type fqdn
        set allow-routing enable
        set fqdn "*.parklogic.com"
    next
end

config router static
    edit 0
        set distance 3
        set blackhole enable
        set dstaddr "wildcard.parklogic.com"
        set vrf 0
    next
end

## ENABLE RE-EVALUATION OF ADDRESSES ON RELOAD OF THE LISTS
config system settings
set ext-resource-session-check enable
end

## CONFIGURE BLOCKLISTS WE HOST ON GITHUB
config system external-resource
    edit "FGT-MB-Domains"
        set type domain
        set category 192
        set resource "https://raw.githubusercontent.com/mb-fgt-lists/threat-actors/refs/heads/main/ta-domain-list.txt"
    next
    edit "FGT-MB-IPs"
        set type address
        set resource "https://raw.githubusercontent.com/mb-fgt-lists/threat-actors/refs/heads/main/ta-ip-list.txt"
    next
end

## CONFIGURE POLICY THAT BLOCKS ANY TRAFFIC TO THE LIST OF IPs
config firewall policy
edit 0
set srcintf any
set dstintf any
set srcaddr all
set dstaddr FGT-MB-IPs 
set schedule always 
set action deny 
set logtraffic disable 
set service ALL
next

## HERE YOU NEED TO LOOKUP THE ID OF THE POLICY (eg. "show")
move [ID OF BLOCK POLICY] before [ID OF FIRST POLICY] # when you hit [TAB] after "before" you get the list of IDs the topmost ist the first
